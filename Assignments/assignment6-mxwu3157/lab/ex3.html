1. Basic scatterplot

Implement a basic scatterplot with the code provided.
<style>
path {
  stroke: black;
}

.tick line {
  stroke: black;
}

.tick {
  stroke: black;
}

.tick text {
  stroke: none;
  fill: black;
}

.dot {
  fill: red;
}

#chart {
	display: inline-block;
	vertical-align: top;
	background-color: #f8f8f8;
}
</style>

<svg id="chart"></svg>
<script src='../node_modules/d3/dist/d3.min.js'></script>

<script>
data = [{ x: 10, y: 8},
        { x: 12, y: 6},
        { x: 24, y: 12}];

var margin = { top: 50, left: 50, bottom: 50, right: 50 };
var width = 400 - margin.left - margin.right;
var height = 400 - margin.top - margin.bottom;

var svg = d3.select('#chart')
  .attr('width', width + margin.left + margin.right)
  .attr('height', height + margin.top + margin.bottom)
  .append('g').attr('transform', 'translate(' + margin.left + ', ' + margin.top + ')');

var y = d3.scaleLinear()
  .domain([6, 12])  //üöß hard-coded values
  .range([height, 0]);

var x = d3.scaleLinear()
  .domain([10, 24])  //üöß hard-coded values
  .range([0, width]);

//plot circles after the axes so they appear on top
svg.selectAll('circle')
  .data(data)
  .enter()
  .append('circle')
  .attr('r', 3)
  .attr('cx', d => x(d.x))
  .attr('cy', d => y(d.y))
  .attr('class', 'dot');

var xAxis = d3.axisBottom()
  .scale(x); //üöß format tick labels

svg.append('g')
  .attr('transform', 'translate(0,' + height + ')')
  .call(xAxis);

var yAxis = d3.axisLeft()
  .scale(y); //üöß format tick labels

svg.append('g')
  .attr('class', 'axis')
  .call(yAxis);
</script>
