<script>
var dataset = [{name: 'A', value: 20},
               {name: 'B', value: 40}];

var width = 200, height = 100;
var dx = 45;  //distance between centers

var svg = d3.select('#ex_3_1')
  .attr('width', width)
  .attr('height', height);

svg.selectAll('circle')
  .data(dataset)
  .enter()
  .append('circle')
  .attr('cx', (d, i) => dx / 2 + i * dx)
  .attr('cy', d => height / 2)
  .attr('r', d => 0.5 * d.value)  //with bubbles map to diameter or area
  .attr('class', 'bubble');

svg.selectAll('text')
  .data(dataset)
  .enter()
  .append('text')
  .attr('x', (d, i) => dx / 2 + i * dx)
  .attr('y', d => height / 2)
  .attr('class', 'bubble')
  .text(d => d.name);

  svg.append('text')
  .attr('x', 10)
  .attr('y', 20)
  .text('Bubble values');
</script>

<script>
var dataset = [[10, 10],
               [20, 50]];

var height = 100, width = 200, pad = 2;  //pad origin!
var svg = d3.select('#ex_3_2')
  .attr('width', width)
  .attr('height', height);

svg.selectAll('circle')
  .data(dataset)
  .enter()
  .append('circle')
  .attr('cx', d => d[0] + pad)
  .attr('cy', d => height - d[1] - pad)  //flip y axis!
  .attr('r', 3)
  .attr('class', 'dot');

//place axes at bottom left corner based on pad 
svg.append('line')
  .attr('x1', pad)
  .attr('y1', height - pad)
  .attr('x2', pad)
  .attr('y2', pad)
  .attr('class', 'axis');

svg.append('line')
  .attr('x1', pad)
  .attr('y1', height - pad)
  .attr('x2', width - pad)
  .attr('y2', height - pad)
  .attr('class', 'axis');

svg.append('text')
  .attr('x', 10)
  .attr('y', 20)
  .text('x vs. y');
</script>
