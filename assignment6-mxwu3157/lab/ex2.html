1. Basic bar chart

Implement a basic bar chart with the code provided.
<style>
path {
  stroke: black;
}

.tick line {
  stroke: black;
}

.tick {
  stroke: black;
}

.tick text {
  stroke: none;
  fill: black;
}

#chart {
  display: inline-block;
  vertical-align: top;
  background-color: #f8f8f8;
}
</style>

<svg id="chart"></svg>

<script src='./node_modules/d3/dist/d3.min.js'></script>
<script src='./node_modules/d3-format/dist/d3-format.min.js'></script>

<script>
var data = [['A', 4500], 
            ['B', 75000], 
            ['C', 130000]];

var margin = { top: 10, right: 20, bottom: 20, left: 40 };
var width = 400 - margin.left - margin.right,
    height = 200 - margin.top - margin.bottom;

var x = d3.scaleBand()  //use scale band for bars
  .domain(['A', 'B', 'C']) //üöß hard-coded values
  .range([0, width])
  .paddingInner(.05); //üöß hard-coded values

var y = d3.scaleLinear()
  .domain([0, 130000]) //üöß hard-coded values
  .range([height, 0]);

var xAxis = d3.axisBottom()
  .scale(x);

var yAxis = d3.axisLeft() //üöß axis labels not formatted 
  .scale(y)
  .ticks(3);  //limit the number of tick marks on y axis

var svg = d3.select("#chart")
  .attr("width", width + margin.left + margin.right)
  .attr("height", height + margin.top + margin.bottom)
  .append("g")
  .attr("transform", "translate(" + margin.left + "," + margin.top + ")");

svg.selectAll('rect')
  .data(data)
  .enter()
  .append('rect')
  .attr('x', d => x(d[0]))
  .attr('y', d => y(d[1]))
  .attr('width', x.bandwidth)
  .attr('height', d => height - y(d[1]))
  .style('fill', 'skyblue');

svg.append("g")
  .attr("transform", "translate(0," + height + ")")
  .call(xAxis);

svg.append("g")
  .call(yAxis);
</script>
