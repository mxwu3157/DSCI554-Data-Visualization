<script>
  var margin = { top: 20, left: 75, bottom: 50, right: 50 },
    width = 850 - margin.left - margin.right,
    height = 350 - margin.top - margin.bottom;

  var svg = d3.select('#chart')
    .append('svg')
    .attr('width', width + margin.left + margin.right)
    .attr('height', height + margin.top + margin.bottom)
    .append('g')
    .attr('transform', 'translate(' + margin.left + ', ' + margin.top + ')');

  var data;
  d3.tsv('planets.tsv', d => {
    return {
      name: d.name,
      gravity: +d.gravity,
      temperature: +d.temperature,
      distance: +d.distance,
    };
  }).then(result => {
    data = result;
    data.sort((a, b) => d3.ascending(a.distance, b.distance));  //sort data in ascending order by distance
    draw();  //draw the bar chart
    toggle('#distance');  //toggle distance button
  });

  ///////////////////////////////////////////////////////////////
  // Controls
  ///////////////////////////////////////////////////////////////

  function toggle(id) {
    d3.selectAll('.button')
      .style('background-color', '#ddd');
    d3.select(id)
      .style('background-color', '#e25a0077');
  }

  //sort by distance button
  d3.select('#distance')
    .on('click', () => {
      data.sort((a, b) => d3.ascending(a.distance, b.distance));  //sort data in ascending order by distance
      x.domain(data.map(d => d.name));
      transition();  //transition the bars
      toggle('#distance');  //set controls
    });

  //sort by temperature button
  d3.select('#temperature')
    .on('click', () => {
      //üöß sort by temperature and transition bars and axis
    });

  //sort by gravity button
  d3.select('#gravity')
    .on('click', () => {
      //üöß sort by gravity and transition bars and axis
    });

  ///////////////////////////////////////////////////////////////
  // axes and transitions
  ///////////////////////////////////////////////////////////////

  var x = d3.scaleBand();
  var y = d3.scaleLinear();

  var xAxis;
  var delay = function (d, i) {
    return i * 50;
  };

  function transition() {
    var transition = svg.transition()  //set-up the transition
      .duration(750)

    transition  //transition the bars
      .selectAll('rect')
      .delay(delay)
      .attr('x', d => x(d.name))

    transition  //transition the axis
      .select('#x-axis')
      .call(xAxis);
  }

  ///////////////////////////////////////////////////////////////
  // draw the chart
  ///////////////////////////////////////////////////////////////

  function draw() {
    x.domain(data.map(d => d.name))
      .range([0, width])
      .paddingInner(0.2);

    y.domain([0, d3.max(data, d => d.gravity)])
      .range([height, 0]);

    svg.selectAll('rect')
      .data(data, d => d.name)
      .enter()
      .append('rect')
      .attr('class', 'bar')
      .attr('x', d => x(d.name))
      .attr('y', d => y(d.gravity))
      .attr('width', x.bandwidth())
      .attr('height', d => height - y(d.gravity));

    xAxis = d3.axisBottom()
      .scale(x);

    svg.append('g')
      .attr('id', 'x-axis')
      .attr('class', 'axis')
      .attr('transform', 'translate(0,' + height + ')')
      .call(xAxis);

    var yAxis = d3.axisLeft()
      .scale(y)
      .ticks(5, 'd');

    svg.append('g')
      .attr('class', 'axis')
      .call(yAxis);

    svg.append('text')
      .attr('x', - height / 2)
      .attr('y', - margin.left * 0.6)
      .attr('class', 'ylabel')
      .append('tspan').text('Gravity (m‚ãÖs')
      .append('tspan').text('-2')
      .attr('class', 'super')
      .append('tspan').text(')');  //üöß fix the last parenthesis in (m.s-2)
  }
</script>
